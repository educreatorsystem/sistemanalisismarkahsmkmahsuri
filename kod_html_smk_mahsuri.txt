        // CSV Parser (Handle quotes & commas)
        function parseCSV(str) {
            const arr = [];
            let quote = false;
            for (let row = 0, col = 0, c = 0; c < str.length; c++) {
                let cc = str[c], nc = str[c+1];
                arr[row] = arr[row] || [];
                arr[row][col] = arr[row][col] || '';
                if (cc == '"' && quote && nc == '"') { arr[row][col] += cc; ++c; continue; }
                if (cc == '"') { quote = !quote; continue; }
                if (cc == ',' && !quote) { ++col; continue; }
                if (cc == '\r' && nc == '\n' && !quote) { ++row; col = 0; ++c; continue; }
                if (cc == '\n' && !quote) { ++row; col = 0; continue; }
                if (cc == '\r' && !quote) { ++row; col = 0; continue; }
                arr[row][col] += cc;
            }
            return arr;
        }

            if (nilai === 'TD') return { gred: 'TD', tafsiran: 'Tidak Ditaksir', mata: 0 };
            
            let markah = parseFloat(nilai);

            return null;
        }

        // Dapatkan hanya rekod TERKINI (menimpa rekod lama jika guru edit)
        function getRekodTerkiniArr(filterFn) {
            let penapis = dataMarkahRaw.filter(filterFn);
            let mapTerkini = {};
            // Oleh kerana rekod terawal di atas dan terkini di bawah, kita timpa key yang sama
            penapis.forEach(rekod => {
                let kunci = rekod.nama + "||" + rekod.subjek;
                mapTerkini[kunci] = rekod;
            });
            return Object.values(mapTerkini);
        }

        function memoriKeMapMarkahTerkini(kls, pen, sub) {
            let arr = getRekodTerkiniArr(r => r.kelas === kls && r.pentaksiran === pen && r.subjek === sub);
            let map = {};
            arr.forEach(a => map[a.nama] = a);
            return map;
        }

        // Pengiraan Purata dan Kumpulan Subjek
        function kumpulDataMurid(arrRekod) {
            let muridMap = {};
            arrRekod.forEach(r => {
                if(!muridMap[r.nama]) {
                    muridMap[r.nama] = { nama: r.nama, kelas: r.kelas, subjekList: {}, jumMarkah: 0, bilSubjekDikira: 0 };
                }
                
                muridMap[r.nama].subjekList[r.subjek] = { markah: r.markah, gred: r.gred };
                
                let m = parseFloat(r.markah);
                if(!isNaN(m)) {
                    muridMap[r.nama].jumMarkah += m;
                    muridMap[r.nama].bilSubjekDikira++;
                }
            });

            // Kira Purata
            let senarai = Object.values(muridMap);
            senarai.forEach(m => {
                if(m.bilSubjekDikira > 0) {
                    m.purata = m.jumMarkah / m.bilSubjekDikira;
                    let gr = kiraGred(m.purata);
                    m.gredPurata = gr ? gr.gred : '-';
                } else {
                    m.purata = 0;
                    m.gredPurata = '-';
                }
            });

            // Susun secara Ranking (Purata Tertinggi -> Terendah)
            senarai.sort((a,b) => b.purata - a.purata);
            return senarai;
        }

                    let terbaik = pelajarSubjek[0]; // ambil top 1
                    tbTerbaik += `<tr>

            if(tbTerbaik === '') tbTerbaik = '<tr><td colspan="5" class="p-4 text-center border border-slate-300">Tiada rekod bernombor.</td></tr>';
            tbodyTerbaik.innerHTML = tbTerbaik;
        }

        function janaPrestasiMP() {
            const tahun = document.getElementById('prestasi-tahun').value;
            const pen = document.getElementById('prestasi-pentaksiran').value;
            const tbody = document.getElementById('jadual-prestasi-body');
            const tfoot = document.getElementById('jadual-prestasi-foot');

            if(!tahun || !pen) { alert("Pilih Tahun dan Pentaksiran."); return; }

            // Cari semua rekod untuk kelas yang bermula dengan Tahun tersebut (cth: "1 AMANAH", "1 BESTARI")
            let rekodTerkini = getRekodTerkiniArr(r => {
                let match = r.kelas.match(/^\d+/);
                return match && match[0] === tahun && r.pentaksiran === pen;
            });

            if(rekodTerkini.length === 0) {
                tbody.innerHTML = `<tr><td colspan="23" class="text-center p-8 text-red-500 border border-slate-300">Tiada data dijumpai.</td></tr>`;
                tfoot.innerHTML = ''; return;
            }

            let subjekUnik = Array.from(new Set(rekodTerkini.map(r => r.subjek))).sort();
            let html = '';
            
            // Pembolehubah Keseluruhan
            let sumDaftar=0, sumAmbil=0, sumTH=0, sumA=0, sumB=0, sumC=0, sumD=0, sumE=0, sumF=0, sumLulus=0, sumGagal=0;
            let totalMata=0;

            subjekUnik.forEach((sub, idx) => {
                let sData = rekodTerkini.filter(r => r.subjek === sub);
                
                let daftar = sData.length;
                let th = sData.filter(r => r.markah === 'TH').length;
                // TD (Tidak Ditaksir) dikira tak ambil exam juga untuk statistik gred, tapi daftar wujud.
                let td = sData.filter(r => r.markah === 'TD').length; 
                let ambil = daftar - th - td;

                let cA = sData.filter(r => r.gred === 'A').length;
                let cB = sData.filter(r => r.gred === 'B').length;
                let cC = sData.filter(r => r.gred === 'C').length;
                let cD = sData.filter(r => r.gred === 'D').length;
                let cE = sData.filter(r => r.gred === 'E').length;
                let cF = sData.filter(r => r.gred === 'F').length;

                let lulus = cA + cB + cC + cD + cE;
                let gagal = cF;

                let p = (v) => ambil > 0 ? ((v/ambil)*100).toFixed(1) : "0.0";
                
                // GPMP: ( (Bil A * 1) + (Bil B * 2) + ... + (Bil F * 6) ) / Bil Ambil
                let nilaiMata = (cA*1) + (cB*2) + (cC*3) + (cD*4) + (cE*5) + (cF*6);
                let gpmp = ambil > 0 ? (nilaiMata / ambil).toFixed(2) : "0.00";

                // Tambah ke Total
                sumDaftar+=daftar; sumAmbil+=ambil; sumTH+=th;
                sumA+=cA; sumB+=cB; sumC+=cC; sumD+=cD; sumE+=cE; sumF+=cF;
                sumLulus+=lulus; sumGagal+=gagal;
                totalMata+=nilaiMata;

                html += `<tr class="hover:bg-slate-50">
                    <td class="px-2 py-1 border border-slate-300 text-center">${idx+1}</td>
                    <td class="px-3 py-1 border border-slate-300 text-left font-semibold">${sub}</td>
                    
                    <td class="px-1 py-1 border border-slate-300 text-center font-medium">${daftar}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center font-medium">${ambil}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center">${th}</td>
                    
                    <td class="px-1 py-1 border border-slate-300 text-center text-green-600">${cA}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-xs text-green-600">${p(cA)}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-emerald-500">${cB}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-xs text-emerald-500">${p(cB)}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-blue-500">${cC}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-xs text-blue-500">${p(cC)}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-amber-500">${cD}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-xs text-amber-500">${p(cD)}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-orange-500">${cE}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-xs text-orange-500">${p(cE)}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-red-600">${cF}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-xs text-red-600">${p(cF)}</td>
                    
                    <td class="px-1 py-1 border border-slate-300 text-center font-bold text-green-700 bg-green-50/50">${lulus}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-xs bg-green-50/50">${p(lulus)}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center font-bold text-red-700 bg-red-50/50">${gagal}</td>
                    <td class="px-1 py-1 border border-slate-300 text-center text-xs bg-red-50/50">${p(gagal)}</td>
                    
                    <td class="px-1 py-1 border border-slate-300 text-center font-bold text-indigo-700 bg-yellow-50/50">${gpmp}</td>
                </tr>`;
            });
            tbody.innerHTML = html;

            // Papar Total Keseluruhan
            let pTotal = (v) => sumAmbil > 0 ? ((v/sumAmbil)*100).toFixed(1) : "0.0";
            let gps = sumAmbil > 0 ? (totalMata / sumAmbil).toFixed(2) : "0.00";

            tfoot.innerHTML = `<tr>
                <td colspan="2" class="px-3 py-2 border border-slate-300 text-right uppercase">Jumlah Keseluruhan</td>
                <td class="px-1 py-2 border border-slate-300 text-center">${sumDaftar}</td>
                <td class="px-1 py-2 border border-slate-300 text-center">${sumAmbil}</td>
                <td class="px-1 py-2 border border-slate-300 text-center">${sumTH}</td>
                
                <td class="px-1 py-2 border border-slate-300 text-center">${sumA}</td><td class="px-1 py-2 border border-slate-300 text-center text-xs">${pTotal(sumA)}</td>
                <td class="px-1 py-2 border border-slate-300 text-center">${sumB}</td><td class="px-1 py-2 border border-slate-300 text-center text-xs">${pTotal(sumB)}</td>
                <td class="px-1 py-2 border border-slate-300 text-center">${sumC}</td><td class="px-1 py-2 border border-slate-300 text-center text-xs">${pTotal(sumC)}</td>
                <td class="px-1 py-2 border border-slate-300 text-center">${sumD}</td><td class="px-1 py-2 border border-slate-300 text-center text-xs">${pTotal(sumD)}</td>
                <td class="px-1 py-2 border border-slate-300 text-center">${sumE}</td><td class="px-1 py-2 border border-slate-300 text-center text-xs">${pTotal(sumE)}</td>
                <td class="px-1 py-2 border border-slate-300 text-center">${sumF}</td><td class="px-1 py-2 border border-slate-300 text-center text-xs">${pTotal(sumF)}</td>
                
                <td class="px-1 py-2 border border-slate-300 text-center text-green-700">${sumLulus}</td><td class="px-1 py-2 border border-slate-300 text-center text-xs">${pTotal(sumLulus)}</td>
                <td class="px-1 py-2 border border-slate-300 text-center text-red-700">${sumGagal}</td><td class="px-1 py-2 border border-slate-300 text-center text-xs">${pTotal(sumGagal)}</td>
                
                <td class="px-1 py-2 border border-slate-300 text-center text-indigo-700">${gps}</td>
            </tr>`;
        }

        // Fungsi ini dikemaskini berdasarkan kod janaPelaporan() baharu
        // dan disesuaikan dengan struktur fail ini: pelaporan-kelas, pelaporan-pentaksiran,
        // cetak-pelaporan serta dataMarkahRaw.
        function janaSlip() {
            const kelasPilihan = document.getElementById('pelaporan-kelas').value;
            const penPilihan = document.getElementById('pelaporan-pentaksiran').value;
            const slipContainer = document.getElementById('cetak-pelaporan');

            if (!kelasPilihan || !penPilihan) {
                alert("Sila pilih Kelas dan Pentaksiran.");
                return;
            }

            // Ambil rekod terkini untuk kelas dan pentaksiran tersebut.
            // getRekodTerkiniArr() memastikan rekod lama ditindih oleh rekod terbaru bagi murid + subjek yang sama.
            const rekodTerkini = getRekodTerkiniArr(item => item.kelas === kelasPilihan && item.pentaksiran === penPilihan);

            // Ambil data murid unik untuk kelas tersebut
            const senaraiMuridInfo = [];
            const muridUnik = new Set();
            rekodTerkini.forEach(item => {
                if (!muridUnik.has(item.nama)) {
                    muridUnik.add(item.nama);
                    senaraiMuridInfo.push({ nama: item.nama, kelas: item.kelas });
                }
            });

            if (senaraiMuridInfo.length === 0) {
                slipContainer.innerHTML = `<div class="bg-white p-8 text-center text-red-500 rounded-xl shadow-sm border border-slate-200 font-bold">Tiada data dijumpai untuk kriteria ini.</div>`;
                return;
            }

            // Kira ranking kelas berdasarkan purata markah sebenar
            let rekodRanking = [];
            senaraiMuridInfo.forEach(murid => {
                let markahMurid = rekodTerkini.filter(m => m.nama === murid.nama);
                let total = 0;
                let count = 0;

                markahMurid.forEach(m => {
                    let val = parseInt(m.markah);
                    if (!isNaN(val)) {
                        total += val;
                        count++;
                    }
                });

                let purata = count > 0 ? (total / count) : 0;
                rekodRanking.push({ nama: murid.nama, purata: purata });
            });

            // Susun dari purata tertinggi ke terendah
            rekodRanking.sort((a, b) => b.purata - a.purata);

            // Berikan kedudukan ranking
            let rankings = {};
            rekodRanking.forEach((rekod, index) => {
                rankings[rekod.nama] = index + 1;
            });

            let slipsHtml = '';

            // Susun murid ikut abjad untuk paparan cetakan slip
            senaraiMuridInfo.sort((a, b) => a.nama.localeCompare(b.nama));

            senaraiMuridInfo.forEach((murid, index) => {
                const markahMurid = rekodTerkini
                    .filter(m => m.nama === murid.nama)
                    .sort((a, b) => a.subjek.localeCompare(b.subjek));

                let jadualMarkahHtml = '';
                let totalMarkah = 0;
                let bilSubjek = 0;
                let statusKeputusan = "LULUS";

                markahMurid.forEach((m, i) => {
                    let markahVal = m.markah;
                    let gredVal = m.gred;

                    if (markahVal !== 'TH' && markahVal !== 'TD' && !isNaN(parseInt(markahVal))) {
                        totalMarkah += parseInt(markahVal);
                        bilSubjek++;
                        if (gredVal === 'F') {
                            statusKeputusan = "GAGAL";
                        }
                    }

                    jadualMarkahHtml += `
                        <tr class="border-b border-gray-300">
                            <td class="p-2 text-center border-r border-gray-300">${i + 1}</td>
                            <td class="p-2 border-r border-gray-300">${m.subjek}</td>
                            <td class="p-2 text-center border-r border-gray-300">${markahVal}</td>
                            <td class="p-2 text-center border-r border-gray-300 font-bold">${gredVal}</td>
                            <td class="p-2 text-center">${m.tafsiran || '-'}</td>
                        </tr>
                    `;
                });

                // Pengiraan Purata dan Gred Keseluruhan
                let nilaiPurata = bilSubjek > 0 ? (totalMarkah / bilSubjek) : 0;
                let purataPaparan = bilSubjek > 0 ? nilaiPurata.toFixed(2) : "0.00";

                // Bundarkan purata untuk menentukan Gred Keseluruhan yang tepat
                let purataBundar = Math.round(nilaiPurata);
                let gredKeseluruhan = '-';

                if (bilSubjek > 0) {
                    if (purataBundar >= 82) gredKeseluruhan = 'A';
                    else if (purataBundar >= 66) gredKeseluruhan = 'B';
                    else if (purataBundar >= 50) gredKeseluruhan = 'C';
                    else if (purataBundar >= 35) gredKeseluruhan = 'D';
                    else if (purataBundar >= 20) gredKeseluruhan = 'E';
                    else gredKeseluruhan = 'F';
                }

                let ulasanGuru = statusKeputusan === "LULUS"
                    ? "Tahniah atas pencapaian yang cemerlang. Teruskan usaha!"
                    : "Tingkatkan usaha anda. Sila tumpukan perhatian pada subjek yang lemah.";

                let kedudukan = rankings[murid.nama] || '-';
                let jumlahMuridKelas = rekodRanking.length;

                slipsHtml += `
                    <div class="slip-page ${index !== senaraiMuridInfo.length - 1 ? 'page-break-after-always' : ''} bg-white p-8 border-b-8 border-gray-100 print:border-none print:p-0">
                        <!-- Header -->
                        <div class="flex items-center justify-center gap-6 mb-6 border-b-2 border-black pb-4">
                            <img src="https://iili.io/CfqiD9p.md.jpg" alt="Logo Sekolah" class="w-20 h-20 object-contain">
                            <div class="text-center">
                                <h1 class="text-lg font-bold uppercase">SEKOLAH MENENGAH KEBANGSAAN MAHSURI</h1>
                                <h2 class="text-lg font-semibold mt-1">SLIP PENCAPAIAN MURID</h2>
                                <p class="text-md mt-1 font-medium">${penPilihan}</p>
                            </div>
                            <div class="w-20"></div>
                        </div>

                        <!-- Maklumat Murid -->
                        <div class="mb-6 w-full flex justify-between font-medium text-sm">
                            <div>
                                <table class="border-none">
                                    <tr><td class="py-1 pr-4">Nama</td><td class="py-1">: <span class="font-bold">${murid.nama}</span></td></tr>
                                    <tr><td class="py-1 pr-4">Kelas</td><td class="py-1">: ${murid.kelas}</td></tr>
                                </table>
                            </div>
                        </div>

                        <!-- Jadual Keputusan Prestasi -->
                        <table class="w-full mb-6 border border-black text-sm">
                            <thead>
                                <tr class="bg-gray-200 border-b border-black">
                                    <th class="p-2 border-r border-black w-10">Bil</th>
                                    <th class="p-2 border-r border-black text-left">Mata Pelajaran</th>
                                    <th class="p-2 border-r border-black w-20">Markah</th>
                                    <th class="p-2 border-r border-black w-16">Gred</th>
                                    <th class="p-2 w-1/3">Tafsiran</th>
                                </tr>
                            </thead>
                            <tbody class="border-b border-black">
                                ${jadualMarkahHtml}
                            </tbody>
                        </table>

                        <!-- Jadual Rumusan Pencapaian -->
                        <div class="mb-6">
                            <h3 class="font-bold text-sm mb-2 uppercase">Rumusan Pencapaian</h3>
                            <table class="w-full border border-black text-sm text-center">
                                <thead>
                                    <tr class="bg-gray-100 border-b border-black">
                                        <th class="p-2 border-r border-black">Bil. MP</th>
                                        <th class="p-2 border-r border-black">Jumlah Markah</th>
                                        <th class="p-2 border-r border-black">Purata</th>
                                        <th class="p-2 border-r border-black">Gred Keseluruhan</th>
                                        <th class="p-2 border-r border-black">Keputusan</th>
                                        <th class="p-2">Kedudukan</th>
                                    </tr>
                                </thead>
                                <tbody>
                                    <tr>
                                        <td class="p-2 border-r border-black font-semibold">${bilSubjek}</td>
                                        <td class="p-2 border-r border-black font-semibold">${totalMarkah}</td>
                                        <td class="p-2 border-r border-black font-semibold">${purataPaparan}</td>
                                        <td class="p-2 border-r border-black font-bold text-lg">${gredKeseluruhan}</td>
                                        <td class="p-2 border-r border-black font-semibold ${statusKeputusan === 'GAGAL' ? 'text-red-600' : ''}">${statusKeputusan}</td>
                                        <td class="p-2 font-semibold">${kedudukan} / ${jumlahMuridKelas}</td>
                                    </tr>
                                </tbody>
                            </table>
                        </div>

                        <!-- Ulasan Guru -->
                        <div class="mb-12 border border-black p-4 text-sm">
                            <p class="font-bold mb-1">Ulasan Guru Kelas:</p>
                            <p class="italic">"${ulasanGuru}"</p>
                        </div>

                        <!-- Tandatangan -->
                        <div class="grid grid-cols-3 gap-8 mt-12 text-sm text-center print:mt-16">
                            <div>
                                <div class="border-b border-black mb-2 h-16"></div>
                                <p class="font-bold">Tandatangan Guru Kelas</p>
                            </div>
                            <div>
                                <div class="border-b border-black mb-2 h-16"></div>
                                <p class="font-bold">Tandatangan Guru Besar</p>
                            </div>
                            <div>
                                <div class="border-b border-black mb-2 h-16"></div>
                                <p class="font-bold">Tandatangan Ibu Bapa/Penjaga</p>
                            </div>
                        </div>
                    </div>
                `;
            });

            slipContainer.innerHTML = slipsHtml;
        }

        // Fungsi Mencetak Global & Ringkas
        function cetakBahagian(idElemen, tajukUtama, subTajuk = '') {
            const kandunganCetak = document.getElementById(idElemen);
            if (!kandunganCetak || kandunganCetak.innerHTML.includes('Sila klik')) {
                alert('Sila jana laporan/analisis terlebih dahulu sebelum mencetak.');
                return;
            }

            const printContent = kandunganCetak.innerHTML;
            const isLandscape = (idElemen === 'cetak-prestasi' || idElemen === 'cetak-individu' || idElemen === 'cetak-kedudukan');
            
            // Tangkap nilai dropdown semasa
            if(idElemen === 'cetak-individu' || idElemen === 'cetak-pelaporan') {
                 subTajuk = document.getElementById(idElemen.split('-')[1]+'-kelas').value + " - " + document.getElementById(idElemen.split('-')[1]+'-pentaksiran').value;
            } else if (idElemen === 'cetak-prestasi' || idElemen === 'cetak-kedudukan') {
                 subTajuk = "Tahun " + document.getElementById(idElemen.split('-')[1]+'-tahun').value + " - " + document.getElementById(idElemen.split('-')[1]+'-pentaksiran').value;
            }

            const printWindow = window.open('', '_blank');
            printWindow.document.write(`
                <!DOCTYPE html>
                <html>
                    <head>
                        <title>Cetak Laporan - SEKOLAH MENENGAH KEBANGSAAN MAHSURI</title>
                        <script src="https://cdn.tailwindcss.com"><\/script>
                        <style>
                            @page { size: ${isLandscape ? 'A4 landscape' : 'A4 portrait'}; margin: 15mm; }
                            body { font-family: 'Arial', sans-serif; -webkit-print-color-adjust: exact; print-color-adjust: exact; background: white; }
                            table { page-break-inside: auto; border-collapse: collapse; width: 100%; }
                            tr { page-break-inside: avoid; page-break-after: auto; }
                            thead { display: table-header-group; }
                            tfoot { display: table-footer-group; }
                            .page-break-after-always { page-break-after: always; }
                            th, td { padding: 4px 6px !important; border: 1px solid #cbd5e1 !important; color: #000 !important; }
                            /* Pastikan header dan tfoot nampak warna bila print */
                            thead th { background-color: #e2e8f0 !important; }
                            tfoot td { background-color: #334155 !important; color: white !important; font-weight: bold;}
                        </style>
                    </head>
                    <body>
                        ${tajukUtama ? `
                        <div class="flex items-center gap-6 mb-6 border-b-2 border-black pb-4">
                            <img src="https://iili.io/CfqiD9p.md.jpg" alt="Logo Sekolah" style="width: 80px; height: 80px; object-fit: contain;">
                            <div>
                                <h1 class="text-xl font-bold uppercase">SEKOLAH MENENGAH KEBANGSAAN MAHSURI</h1>
                                <h2 class="text-lg font-semibold text-gray-800">${tajukUtama}</h2>
                                <p class="text-md font-medium text-gray-600 mt-1">${subTajuk}</p>
                            </div>
                        </div>` : ''}
                        
                        ${printContent}
                        
                        <script>
                            window.onload = function() {
                                setTimeout(() => { window.print(); window.close(); }, 500);
                            };
                        <\/script>
                    </body>
                </html>
            `);
            printWindow.document.close();
        }

        // UI Modal Helpers
        function bukaModal(mesej, type) {
            const m = document.getElementById('modal-global');
            const c = document.getElementById('modal-content');
            m.classList.remove('hidden');
            if(type === 'loading') {
                c.innerHTML = `<div class="loader mb-4"></div><p class="font-medium text-slate-700">${mesej}</p>`;
            } else if(type === 'success') {
                c.innerHTML = `<div class="text-green-500 mb-2"><svg class="w-12 h-12 mx-auto" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M5 13l4 4L19 7"></path></svg></div><p class="font-bold text-slate-800">${mesej}</p>`;
            } else {
                c.innerHTML = `<div class="text-red-500 mb-2"><svg class="w-12 h-12 mx-auto" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"></path></svg></div><p class="font-bold text-slate-800">${mesej}</p>`;
            }
        }
        function tutupModal() { document.getElementById('modal-global').classList.add('hidden'); }
